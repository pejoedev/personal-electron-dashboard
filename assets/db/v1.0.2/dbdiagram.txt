Table feed {
  uuid TEXT [pk]
  name TEXT
  link TEXT
  rss_url TEXT
  last_fetch TEXT
  description TEXT
  language TEXT
}

Table message {
  uuid TEXT [pk]
  title TEXT
  link TEXT
  description TEXT
  publication_date TEXT
  viewed BOOLEAN
}

Table rss {
  uuid TEXT [pk]
  feedId TEXT [ref: > feed.uuid]
  messageId TEXT [ref: > message.uuid]
}

Table project {
  uuid TEXT [pk]
  name TEXT
  link TEXT // project link for navigating to it
  healthCheckUrl TEXT // if empty string, do not do healthcare
}

Table securityAlert {
  uuid TEXT [pk]
  messageId TEXT [ref: > message.uuid]
  projectId TEXT [ref: > project.uuid]
  level TEXT
  blob TEXT
  resolved BOOLEAN
}

Table projectHealthCheck {
  uuid TEXT [pk]
  projectId TEXT [ref: > project.uuid]
  fetch TEXT
  status TEXT
  httpResponse TEXT
}
