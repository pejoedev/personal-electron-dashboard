Table feed {
  uuid TEXT [pk]
  name TEXT
  link TEXT
  description TEXT
  language TEXT
}

Table message {
  uuid TEXT [pk]
  title TEXT
  link TEXT
  description TEXT
  pubDate TEXT
  viewed BOOLEAN
}

Table rss {
  uuid TEXT [pk]
  feedId TEXT [ref: > feed.uuid]
  messageId TEXT [ref: > message.uuid]
}

Table project {
  uuid TEXT [pk]
  name TEXT
  link TEXT
}

Table securityAlert {
  uuid TEXT [pk]
  messageId TEXT [ref: > message.uuid]
  projectId TEXT [ref: > project.uuid]
  level TEXT
  blob TEXT
  resolved BOOLEAN
}
